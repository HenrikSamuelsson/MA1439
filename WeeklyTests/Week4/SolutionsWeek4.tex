\section*{Uppgift 1.}
För vilken exponent $x$ är
\begin{enumerate}[(a)]
\item $3^x \cdot 3^{-2}=\dfrac{3^3}{3}$
\item $7^x + 5 \cdot 7^x = 294$
\end{enumerate}

\textbf{Lösning}
\begin{enumerate}[(a)]
\item $ 3^x \cdot 3^{-2} = \dfrac{3^3}{3} $

$ 3^x \cdot 3^{-2} = 3^2 $

$ 3^x = 3^3 \cdot 3^2 $

$ 3^x = 3^{3 + 2} $

$ 3^x  = 3^5 $

$ x = 5 $
\item $ 7^x + 5 \cdot 7^x = 294 $

$ 7^x(1 + 5) = 294 $

$ 7^x \cdot 6 = 294 $

$ 7^x = 294 / 6 $

$ 7^x = 49 $

$ 7^x = 7 \cdot 7 $

$ 7^x = 7^2 $

$ x = 2 $
\end{enumerate}
